CREATE TABLE usuario(
    usuario_id SERIAL PRIMARY KEY,
    nombre_usuario VARCHAR(50) UNIQUE NOT NULL,
    contrasena_usuario VARCHAR(255) NOT NULL
);

CREATE TABLE electrodomesticos(
    id_electro SERIAL PRIMARY KEY,
    usuario_id INTEGER REFERENCES usuario(usuario_id),
    nombre VARCHAR(100) NOT NULL,
    potencia_w DECIMAL(10,2) NOT NULL,
    horas_dia DECIMAL(5,2) NOT NULL,
    cantidad INTEGER NOT NULL
);

CREATE TABLE calculos_historial (
    id_calculo SERIAL PRIMARY KEY,
    usuario_id INTEGER REFERENCES usuario(usuario_id) ON DELETE CASCADE,
    mes INTEGER NOT NULL CHECK (mes BETWEEN 1 AND 12),
    anio INTEGER NOT NULL CHECK (anio >= 2000),
    consumo_diario DECIMAL(10,2) NOT NULL,
    consumo_mensual DECIMAL(10,2) NOT NULL,
    costo_mensual DECIMAL(10,2) NOT NULL,
    precio_kwh DECIMAL(10,2) NOT NULL
);

CREATE TABLE detalle_calculo (
    id_detalle SERIAL PRIMARY KEY,
    id_calculo INTEGER REFERENCES calculos_historial(id_calculo) ON DELETE CASCADE,
    electrodomestico_nombre VARCHAR(100) NOT NULL,
    potencia_w DECIMAL(10,2) NOT NULL,
    horas_dia DECIMAL(5,2) NOT NULL,
    cantidad INTEGER NOT NULL,
    consumo_diario DECIMAL(10,2) NOT NULL,
    consumo_mensual DECIMAL(10,2) NOT NULL,
    costo_mensual DECIMAL(10,2) NOT NULL
);
